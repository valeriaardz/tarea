# Importar bibliotecas necesarias
import pandas as pd
import numpy as np
from google.colab import files
import statsmodels.api as sm
from statsmodels.tsa.stattools import adfuller
from statsmodels.tsa.arima.model import ARIMA
import matplotlib.pyplot as plt

# Funcin para prueba de raz unitaria (ADF) y estacionariedad
def unit_root_test(series, name):
    print(f"\nPrueba de Raz Unitaria (ADF) para {name}:")
    adf_result = adfuller(series)
    print(f'Estadstico ADF: {adf_result[0]:.4f}')
    print(f'p-valor: {adf_result[1]:.4f}')
    print(f"Valores crticos: 1%={adf_result[4]['1%']:.4f}, 5%={adf_result[4]['5%']:.4f}, 10%={adf_result[4]['10%']:.4f}")
    if adf_result[1] < 0.05:
        print("Resultado: Estacionaria (rechazo H0: hay raz unitaria)")
    else:
        print("Resultado: No estacionaria (no rechazo H0: hay raz unitaria)")
    return adf_result[1]

# Funcin para encontrar el mejor modelo ARIMA
def find_best_arima(series, max_p=2, max_d=1, max_q=2):
    best_aic = float('inf')
    best_order = None
    for p in range(max_p + 1):
        for d in range(max_d + 1):
            for q in range(max_q + 1):
                try:
                    model = ARIMA(series, order=(p, d, q))
                    results = model.fit()
                    if results.aic < best_aic:
                        best_aic = results.aic
                        best_order = (p, d, q)
                except:
                    continue
    print(f"\nMejor ARIMA para la serie: {best_order}, AIC: {best_aic:.2f}")
    return best_order

for series, label in temporalities:
    print(f"\n=== Anlisis para {label} ===")

    # Prueba de raz unitaria y estacionariedad
    p_value = unit_root_test(series, label)

    # Diferenciar si no es estacionaria
    if p_value >= 0.05:
        print(f"\nLa serie no es estacionaria. Diferenciando una vez...")
        series_diff = series.diff().dropna()
        p_value_diff = unit_root_test(series_diff, f"{label} Diferenciada")
        if p_value_diff < 0.05:
            print("La serie diferenciada es estacionaria. Usaremos d=1 en ARIMA.")
            series_to_model = series  # Usamos la serie original, pero ARIMA aplicar d=1
        else:
            print("La serie diferenciada an no es estacionaria. Usaremos d=2 en ARIMA.")
            series_to_model = series  # ARIMA aplicar d=2
    else:
        print("La serie ya es estacionaria. Usaremos d=0 en ARIMA (es decir, un modelo ARMA).")
        series_to_model = series

    # Ajustar modelo ARIMA
    arima_order = find_best_arima(series_to_model)
    arima_model = ARIMA(series_to_model, order=arima_order).fit()

    # Pronstico: 60 perodos (1 hora para 1 min, 5 horas para 5 min)
    forecast_steps = 60

    plt.title(f'Pronstico de Precio - {label}')
    plt.xlabel('Hora (EST, 11 de marzo de 2025)')
    plt.ylabel('Precio')
    plt.legend()

    plt.show()

    print(f"ltimo precio: {last_price:.2f}")
    print(f"Prediccin promedio: {mean_forecast:.2f}")
    print(f"Intervalo 95% al final: [{conf_int.iloc[-1, 0]:.2f}, {conf_int.iloc[-1, 1]:.2f}]")

        "Hora (EST)": time_labels,
        "Prediccin": forecast.values,
        "Lmite Inferior 95%": conf_int.iloc[:, 0],
        "Lmite Superior 95%": conf_int.iloc[:, 1]
    })

    print(forecast_df)
